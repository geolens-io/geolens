**1. Create `frontend/src/components/auth/AuthPrompt.tsx`:**

A small reusable inline login prompt component. IMPORTANT: use `useTranslation('auth')` (the 'auth' namespace) — NOT the default namespace. All auth components in this project use `useTranslation('auth')` with flat keys like `t('signIn')`.
```tsx
import { Link, useLocation } from 'react-router';
import { useTranslation } from 'react-i18next';
import { LogIn } from 'lucide-react';

export function AuthPrompt({ action }: { action: string }) {
  const location = useLocation();
  const { t } = useTranslation('auth');
  return (
    <Link
      to="/login"
      state={{ from: location.pathname }}
      className="inline-flex items-center gap-1.5 text-sm text-primary hover:underline"
    >
      <LogIn className="h-3.5 w-3.5" />
      {t('signInTo', { action })}
    </Link>
  );
}
```
This preserves post-login redirect via `location.state.from` (existing login redirect machinery handles this — per Phase quick-260323-dxj).
