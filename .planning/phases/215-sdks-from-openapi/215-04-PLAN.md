    class TestPythonAuthWrapperUnit:

    @pytest.mark.asyncio
    async def test_typescript_round_trip(client, admin_auth_header) -> None:
        """Spawn a Node subprocess that exercises the TypeScript SDK against
        a session-scoped uvicorn instance.

        Per RESEARCH Assumption A3: if `node` is not available on the runner
        OR if the SDK has not been built (`dist/index.js` missing), skip with
        a clear reason. The CI workflow ensures both are present (Plan 04
        sub-step in ci.yml installs node + runs `npm install && npm run build`
        before the round-trip step).
        """
        node = shutil.which("node")
        if not node:
            pytest.skip("node not available on this runner")

        ts_dist = REPO_ROOT / "sdks" / "typescript" / "dist" / "index.js"
        if not ts_dist.exists():
            pytest.skip(
                f"TypeScript SDK not built ({ts_dist} missing). "
                "Run `cd sdks/typescript && npm install && npm run build` first."
            )

        # The TS SDK can't use ASGITransport — it needs a real HTTP server.
        # Spin up a uvicorn-like in-process server using httpx + an asgi
        # adapter is too clever; the simplest stable approach is to spawn a
        # uvicorn subprocess for this single test, then tear it down.
        # For Phase 215 v13.1, we accept the 1-2s startup cost (per Pitfall 8
        # tradeoff) and do the spawn inline.
        #
        # Avoid binding to a fixed port to prevent CI conflicts: bind 127.0.0.1:0
        # via a uvicorn config object, then read the resulting address from the
        # server's `started` event.
        #
        # Skip if uvicorn unavailable.
        try:
            import uvicorn  # noqa: F401
        except ImportError:
            pytest.skip("uvicorn not available (test needs a real HTTP server for TS half)")

        import asyncio
        import socket

        from app.api.main import app

        # Find a free port (race window is acceptable for a one-shot test)
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.bind(("127.0.0.1", 0))
        port = sock.getsockname()[1]
        sock.close()

        config = uvicorn.Config(app, host="127.0.0.1", port=port, log_level="error")
        server = uvicorn.Server(config)
        task = asyncio.create_task(server.serve())
        # Wait for server-up
        for _ in range(30):
            await asyncio.sleep(0.1)
            if server.started:
                break
        else:
            task.cancel()
            pytest.fail("uvicorn server did not start within 3s")

        try:
            base_url = f"http://127.0.0.1:{port}"
            token = admin_auth_header["Authorization"].removeprefix("Bearer ")
            env = os.environ.copy()
            env["GEOLENS_BASE_URL"] = base_url
            env["GEOLENS_TOKEN"] = token

            ts_test = REPO_ROOT / "sdks" / "typescript" / "test" / "round_trip.test.mjs"
            assert ts_test.exists(), f"TS test script missing at {ts_test}"

            result = subprocess.run(
                [node, str(ts_test)],
                env=env,
                cwd=REPO_ROOT / "sdks" / "typescript",
                capture_output=True,
                text=True,
                timeout=30,
            )
            assert result.returncode == 0, (
                f"TS round-trip failed (exit {result.returncode}):\n"
                f"STDOUT:\n{result.stdout}\n"
                f"STDERR:\n{result.stderr}"
            )
        finally:
            server.should_exit = True
            try:
                await asyncio.wait_for(task, timeout=5)
            except (asyncio.TimeoutError, asyncio.CancelledError):
                task.cancel()

    // on the generator's camelcasing of the FastAPI operationId. Common shapes:
    //   searchDatasetsEndpointSearchDatasetsGet
    //   getSingleDatasetDatasetsDatasetIdGet
    //   uploadFileIngestUploadPost
    // We import them lazily so build failures point to the exact missing export.

    const token = process.env.GEOLENS_TOKEN;

    function assert(cond, msg) {
      if (!cond) {
        console.error('FAIL:', msg);

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| GitHub Actions runner → external registries | `uv publish` (PyPI) and `npm publish` (npm) ship signed/authenticated packages. Tokens flow through GitHub repo secrets — never logged, never committed. |
| pytest → in-process FastAPI app (Python half) | ASGITransport keeps the Python round-trip purely in-memory; no network namespace, no shared port, no test pollution risk. |
| pytest → Node subprocess + uvicorn (TS half) | The TypeScript half binds to a free port on 127.0.0.1 only; the subprocess inherits a sanitized env (`os.environ.copy()` plus injected `GEOLENS_TOKEN`); subprocess is killed in `finally` to prevent runaway servers. |
| publish-sdks.yml → user-managed tokens | Tokens are scoped per-package (PyPI: project-scoped; npm: package-scoped). Manual-trigger workflow + dry_run option lets the user verify before pressing publish. |
