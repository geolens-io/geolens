```bash
#!/usr/bin/env bash
set -euo pipefail
cd "$(dirname "$0")/.."

echo "Asserting canonical URL points to docs.getgeolens.com..."
grep -F 'rel="canonical"' dist/index.html | grep -F 'https://docs.getgeolens.com' \
  || { echo "FAIL: canonical missing or wrong"; exit 1; }

echo "Asserting noindex meta present..."
grep -F 'name="robots"' dist/index.html | grep -F 'noindex' \
  || { echo "FAIL: noindex meta missing"; exit 1; }

echo "Asserting sitemap-index.xml generated..."
test -f dist/sitemap-index.xml || { echo "FAIL: sitemap-index missing"; exit 1; }

echo "Asserting _redirects copied to dist/..."
test -f dist/_redirects || { echo "FAIL: _redirects missing"; exit 1; }

echo "Asserting _redirects contains expected three-rule pattern..."
expected_301_count=$(grep -c '301$' dist/_redirects || echo 0)
if [ "$expected_301_count" -lt 9 ]; then
  echo "FAIL: expected at least 9 '301' rules in _redirects, got $expected_301_count"
  exit 1
fi

echo "Asserting NO flat URLs (/install, /admin, /api) in built HTML internal links..."
# Catches accidental flat URLs in links/sidebar; /quickstart references are fine
# (those go to marketing site, owned by HeroSection.astro/Nav.astro).
if grep -rE 'href="/(install|admin|api)(/|"|#)' dist/ --include='*.html' 2>/dev/null; then
  echo "FAIL: flat URL detected in built HTML"
  exit 1
fi

echo "Asserting Disallow: / present in robots.txt..."
grep -F 'Disallow: /' dist/robots.txt \
  || { echo "FAIL: robots.txt missing Disallow: /"; exit 1; }

echo "All build-artifact assertions passed."
```
