// FeatureStripe.astro
interface Props {
  eyebrow: string;       // Uppercase tracking-widest label above heading (e.g., "Search & Discover")
  heading: string;       // h2 — the capability name / outcome (e.g., "Find any dataset in seconds")
  body: string;          // Description paragraph
  bullets?: string[];    // Optional key points (3-5 max) displayed below body
  previewLeft?: boolean; // false (default) = copy left, preview right; true = flipped
  background?: 'default' | 'surface'; // 'default' = var(--background) white, 'surface' = var(--surface-0) pale
}
const {
  eyebrow,
  heading,
  body,
  bullets = [],
  previewLeft = false,
  background = 'default',
} = Astro.props;

const bgStyle = background === 'surface'
  ? 'background-color: var(--surface-0);'
  : 'background-color: var(--background);';
--- <!-- astro frontmatter fence -->

<section
  class="w-full overflow-x-clip py-20 sm:py-28 px-4 sm:px-6 lg:px-8"
  style={bgStyle}
>
  <div class="max-w-7xl mx-auto grid grid-cols-1 lg:grid-cols-12 gap-12 lg:gap-16 items-center">
    <!-- Copy column -->
    <div class:list={["lg:col-span-5", previewLeft && "lg:order-2"]}>
      <p class="text-base font-semibold uppercase tracking-widest" style="color: var(--primary-700);">{eyebrow}</p>
      <h2 class="mt-3 text-3xl font-semibold leading-[1.2]" style="color: var(--foreground);">{heading}</h2>
      <p class="mt-4 text-base" style="color: var(--muted-foreground); line-height: 1.5;">{body}</p>
      {bullets.length > 0 && (
        <ul class="mt-6 space-y-2">
          {bullets.map((b) => (
            <li class="flex gap-2 text-sm" style="color: var(--foreground);">
              <span aria-hidden="true" style="color: var(--primary-700);">•</span>
              <span>{b}</span>
            </li>
          ))}
        </ul>
      )}
    </div>
    <!-- Preview column with w-full max-w-md wrapper (Phase 215 Plan 04 responsive pattern) -->
    <div class:list={["lg:col-span-7 flex justify-center", previewLeft && "lg:order-1"]}>
      <div class="w-full max-w-md mx-auto">
        <slot />
      </div>
    </div>
  </div>
</section>
