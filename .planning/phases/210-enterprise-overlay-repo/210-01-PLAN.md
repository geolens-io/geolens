```python
"""GeoLens Enterprise extensions package."""
from __future__ import annotations


def register_extensions(registry: dict) -> None:
    """Register all enterprise extensions with the core registry.

    Called by core load_extensions() via entry_points discovery.
    """
    # Protocol implementations (placeholder until Plan 02 extracts features)
    registry["auth"] = _get_auth_extension()
    registry["audit"] = _get_audit_extension()
    registry["branding"] = _get_branding_extension()

    # Collect routers for dynamic registration
    registry["_routers"] = _get_routers()


def _get_auth_extension():
    from geolens_enterprise.auth.saml import EnterpriseSamlExtension
    return EnterpriseSamlExtension()


def _get_audit_extension():
    from geolens_enterprise.audit import EnterpriseAuditExtension
    return EnterpriseAuditExtension()


def _get_branding_extension():
    from geolens_enterprise.branding import EnterpriseBrandingExtension
    return EnterpriseBrandingExtension()


def _get_routers() -> list:
    """Return FastAPI routers to register dynamically."""
    routers = []
    try:
        from geolens_enterprise.auth.saml.router import router as saml_router
        routers.append(saml_router)
    except ImportError:
        pass
    try:
        from geolens_enterprise.audit.export import router as audit_export_router
        routers.append(audit_export_router)
    except ImportError:
        pass
    try:
        from geolens_enterprise.branding.router import router as branding_router
        routers.append(branding_router)
    except ImportError:
        pass
    return routers


def get_migration_paths() -> list[str]:
    """Return paths to enterprise Alembic migration directories."""
    import importlib.resources
    return [str(importlib.resources.files("geolens_enterprise") / "migrations" / "versions")]
```

**1. Update `backend/alembic/env.py` for multi-directory migration discovery (per D-01, D-02).**

Add a `_discover_migration_paths()` function BEFORE the `do_run_migrations` function. This must run before `context.configure()` to avoid the memoization issue (Pitfall 2 from research).

Insert after `config = context.config` and `fileConfig(...)` but BEFORE `target_metadata`:

```python
import pathlib
from importlib.metadata import entry_points as iter_entry_points

def _discover_migration_paths() -> list[str]:
    """Discover additional migration version directories from plugins."""
    paths = []
    for ep in iter_entry_points(group="geolens.migrations"):
        try:
            fn = ep.load()
            if callable(fn):
                for p in fn():
                    if pathlib.Path(p).is_dir():
                        paths.append(p)
        except Exception:
            pass  # Non-fatal: core migrations still run
    return paths

# Append enterprise migration paths to version_locations
_extra_paths = _discover_migration_paths()
if _extra_paths:
    _base_versions = config.get_main_option("version_locations") or "alembic/versions"
    _all_paths = _base_versions + " " + " ".join(_extra_paths)
    config.set_main_option("version_locations", _all_paths)
```

Keep the rest of env.py unchanged.
