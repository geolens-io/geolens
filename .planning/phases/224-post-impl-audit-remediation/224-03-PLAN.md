   **If using Starlette's BaseHTTPMiddleware:**
   The cleanest approach is to convert to raw ASGI middleware for stream wrapping control:
   ```python
   class RequestBodyLimitMiddleware:
       def __init__(self, app, max_bytes: int):
           self.app = app
           self.max_bytes = max_bytes

       async def __call__(self, scope, receive, send):
           if scope["type"] != "http":
               await self.app(scope, receive, send)
               return

           # Fast path: Content-Length header check
           headers = dict(scope.get("headers", []))
           content_length = headers.get(b"content-length")
           if content_length is not None:
               if int(content_length) > self.max_bytes:
                   response = Response("Request body too large", status_code=413)
                   await response(scope, receive, send)
                   return

           # Stream-counting wrapper for chunked requests
           total_read = 0
           max_bytes = self.max_bytes

           async def limited_receive():
               nonlocal total_read
               message = await receive()
               if message.get("type") == "http.request":
                   body = message.get("body", b"")
                   total_read += len(body)
                   if total_read > max_bytes:
                       raise HTTPException(status_code=413, detail="Request body too large")
               return message

           await self.app(scope, limited_receive, send)
   ```

3. **Wrap `recover_stale_jobs` in advisory lock** in `backend/app/worker.py`:
   ```python
   from sqlalchemy import text

   RECOVERY_LOCK_KEY = 224_001  # stable app-unique integer

   async def recover_stale_jobs():
       async with async_session() as session:
           # Advisory lock prevents concurrent recovery across workers
           lock_result = await session.execute(
               text("SELECT pg_try_advisory_xact_lock(:key)"),
               {"key": RECOVERY_LOCK_KEY}
           )
           if not lock_result.scalar():
               logger.info("Stale job recovery skipped — another worker holds the lock")
               return

           # Only mark jobs as stale if their heartbeat is older than 5 minutes
           # (or they have no heartbeat at all AND were created > 10 minutes ago)
           stale_cutoff = datetime.now(timezone.utc) - timedelta(minutes=5)
           stale_query = select(IngestJob).where(
               IngestJob.status == "running",
               or_(
                   IngestJob.last_heartbeat_at < stale_cutoff,
                   and_(
                       IngestJob.last_heartbeat_at.is_(None),
                       IngestJob.created_at < stale_cutoff,
                   ),
               ),
           )

           result = await session.execute(stale_query)
           stale_jobs = result.scalars().all()

           for job in stale_jobs:
               job.status = "failed"
               job.error = "Marked as stale by recovery process (no heartbeat)"
               logger.warning("Marking stale job as failed",
                             job_id=str(job.id), last_heartbeat=str(job.last_heartbeat_at))

           await session.commit()
   ```
