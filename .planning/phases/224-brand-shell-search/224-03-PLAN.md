    ```javascript
    // ec-pagefind-weight.mjs — SEARCH-02 / D-28.
    //
    // Adds data-pagefind-weight="0.1" to every Expressive Code rendered code block.
    // Per RESEARCH.md Pivot #1: a rehype plugin DOES NOT work — EC replaces <pre> AST nodes
    // with its own rendered <figure><pre> tree, discarding any attributes a user-defined
    // rehype plugin sets. The postprocessRenderedBlock hook runs AFTER EC's render and is
    // the only mechanism that survives to the built HTML.
    //
    // Pagefind weight semantics (https://pagefind.app/docs/weighting/):
    //   - default 1.0
    //   - 0.1 = code stays findable but ranks at 1/100 of normal prose (quadratic scaling)
    //   - per CONTEXT.md D-28: 0.1 (NOT 0.5 — milestone forecasts heavy code content)
    //
    // Wired in docs/astro.config.mjs:
    //   starlight({ expressiveCode: { plugins: [pluginPagefindWeight()] } })
    import { definePlugin } from '@expressive-code/core';

    export function pluginPagefindWeight() {
      return definePlugin({
        name: 'pagefind-weight',
        hooks: {
          postprocessRenderedBlock: (context) => {
            const { blockAst } = context.renderData;
            blockAst.properties = blockAst.properties || {};
            blockAst.properties['data-pagefind-weight'] = '0.1';
          },
        },
      });
    }
    ```

    The path is `docs/plugins/ec-pagefind-weight.mjs` (NOT `docs/plugins/rehype-pagefind-code-weight.mjs` — that's the abandoned UI-SPEC name; Pivot #1 renamed it). Create the `plugins/` directory if it doesn't exist.
