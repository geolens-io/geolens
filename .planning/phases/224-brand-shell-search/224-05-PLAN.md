    #!/usr/bin/env node
    /*
     * verify-shell-layout.mjs — SHELL-05 runtime non-overlap probe (gap-closure plan 224-05).
     *
     * Drives Playwright Chromium against a locally-running `npm run preview` server to
     * assert that the back-link and the Starlight site-title anchors do NOT overlap in
     * either light or dark mode at desktop (1280×800) and mobile (360×800) widths.
     *
     * Static grep cannot prove non-overlap — bounding-rect math requires a real layout
     * engine. We use Playwright Chromium because (a) playwright@1.59.1 is already in
     * /Users/ishiland/Code/getgeolens.com/node_modules (root devDependency); (b) layout
     * is not a browser-engine bug — single-engine probe is sufficient.
     *
     * Usage:
     *   # Terminal 1: cd docs && npm run preview
     *   # Terminal 2: cd docs && node scripts/verify-shell-layout.mjs
     *
     * Exit 0 on success; exit 1 with bounding-rect diff on overlap or unreachable preview.
     */

    import { chromium } from 'playwright';

    const PREVIEW_URL = 'http://localhost:4321/';
    const VIEWPORTS = [
      { name: 'desktop', width: 1280, height: 800 },
      { name: 'mobile',  width: 360,  height: 800 },
    ];
    const MODES = ['light', 'dark'];
    const EPSILON = 0.5;

    function rectsIntersect(a, b) {
      return !(
        a.right + EPSILON <= b.left ||
        b.right + EPSILON <= a.left ||
        a.bottom + EPSILON <= b.top ||
        b.bottom + EPSILON <= a.top
      );
    }

    async function probe(page, viewport, mode) {
      await page.setViewportSize({ width: viewport.width, height: viewport.height });
      // Set theme via localStorage then navigate so Starlight picks it up on boot.
      await page.addInitScript((m) => {
        localStorage.setItem('starlight-theme', m);
      }, mode);
      await page.goto(PREVIEW_URL, { waitUntil: 'domcontentloaded' });
      await page.waitForSelector('a.back-link', { timeout: 5000 });
      await page.waitForSelector('.header .title-wrapper', { timeout: 5000 });

      const rects = await page.evaluate(() => {
        const backLink = document.querySelector('a.back-link');
        // Starlight SiteTitle renders an <a> inside .title-wrapper. Match anchors INSIDE
        // .title-wrapper but exclude the back-link itself (defensive — back-link is a sibling
        // of .title-wrapper, not a descendant, but exclusion keeps the selector self-evident).
        const titleAnchor =
          document.querySelector('.header .title-wrapper a:not(.back-link)') ||
          document.querySelector('.header .site-title a') ||
          document.querySelector('.header .title-wrapper');
        const toRect = (el) => {
          const r = el.getBoundingClientRect();
          return { x: r.x, y: r.y, w: r.width, h: r.height,
                   left: r.left, right: r.right, top: r.top, bottom: r.bottom };
        };
        return {
          backLink: backLink ? toRect(backLink) : null,
          siteTitle: titleAnchor ? toRect(titleAnchor) : null,
        };
      });

    }

      } finally {

      }

        process.exit(0);
      } else {

      }
    }
