    ```bash
    #!/usr/bin/env bash
    # check-token-sync.sh — BRAND-04 runtime enforcement.
    #
    # Asserts that the OKLCH primary palette in:
    #   getgeolens.com/src/styles/global.css      (marketing source-of-truth)
    # matches:
    #   getgeolens.com/docs/src/styles/custom.css (docs mirror)
    # for stops 50, 100, 200, 300, 400, 500, 600, 700, 800, 900.
    #
    # Stop 950 is extrapolated for docs-only and intentionally not synced (D-02).
    # Phase 227 (CONTRIBUTING.md update) documents the maintenance convention prose-side (D-08).
    #
    # Wired into docs-ci.yml between `npm ci` and `npx astro check` (D-07, D-34).
    # Failing this script fails the CI build (and therefore the deploy).

    set -euo pipefail

    SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
    DOCS_CSS="$SCRIPT_DIR/../src/styles/custom.css"
    MARKETING_CSS="$SCRIPT_DIR/../../src/styles/global.css"

    test -f "$DOCS_CSS"      || { echo "FAIL: docs custom.css not found at $DOCS_CSS"; exit 1; }
    test -f "$MARKETING_CSS" || { echo "FAIL: marketing global.css not found at $MARKETING_CSS"; exit 1; }

    STOPS=(50 100 200 300 400 500 600 700 800 900)
    errors=0

    for stop in "${STOPS[@]}"; do
      marketing_line=$(grep -E "^\s*--primary-${stop}:" "$MARKETING_CSS" | head -1)
      docs_line=$(grep -E "^\s*--primary-${stop}:" "$DOCS_CSS" | head -1)

      marketing_oklch=$(echo "$marketing_line" | grep -oE 'oklch\([^)]+\)' | tr -s ' ')
      docs_oklch=$(echo "$docs_line" | grep -oE 'oklch\([^)]+\)' | tr -s ' ')

      if [ -z "$marketing_oklch" ]; then
        echo "FAIL: --primary-${stop} not found in marketing global.css"
        errors=$((errors+1))
        continue
      fi
      if [ -z "$docs_oklch" ]; then
        echo "FAIL: --primary-${stop} not found in docs custom.css"
        errors=$((errors+1))
        continue
      fi
      if [ "$marketing_oklch" != "$docs_oklch" ]; then
        echo "FAIL: --primary-${stop} drift detected"
        echo "  marketing: $marketing_oklch"
        echo "  docs:      $docs_oklch"
        errors=$((errors+1))
      fi
    done

    if [ "$errors" -gt 0 ]; then
      echo ""
      echo "Token drift detected. Reconcile docs/src/styles/custom.css against"
      echo "getgeolens.com/src/styles/global.css :root --primary-* declarations."
      exit 1
    fi

    echo "All 10 --primary-* stops in sync between marketing and docs."
    ```
