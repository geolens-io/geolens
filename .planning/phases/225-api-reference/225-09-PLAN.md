```typescript
// pagefind-exclude.ts — D-21 / D-22 (Phase 225 API Reference, success criterion #4).
//
// Starlight route middleware that excludes the auto-generated starlight-openapi
// reference subtree from the Pagefind search index. Mechanism: mutate
// `starlightRoute.entry.data.pagefind = false` for any URL under
// `/guides/api/operations/` — Starlight's Page.astro emits `data-pagefind-body`
// on <main> only when that flag is !== false, and Pagefind only indexes pages
// with that attribute. Verified against:
//   - withastro/starlight @ packages/starlight/components/Page.astro
//   - https://starlight.astro.build/guides/route-data/
//   - https://pagefind.app/docs/indexing/
//
// Why this mechanism (vs alternatives):
//   - starlight-openapi 0.25.0 does NOT support frontmatter passthrough, so
//     `pagefind: false` in MDX won't work for plugin-rendered pages.
//   - `data-pagefind-ignore` on a wrapper conflicts with Starlight's
//     `data-pagefind-body` model.
//   - Pagefind config `exclude_selectors` is CSS-selector-based — there is no
//     URL/path glob option (verified at pagefind.app/docs/config-options/).
//   - Post-build deletion of fragment files races with Pagefind's index format.
// routeMiddleware is the only clean, supported path.
//
// URL space (researcher-verified by reading
// packages/starlight-openapi/libs/route.ts):
//   - `/guides/api/`                              → schema overview
//                                                   (our hand-authored
//                                                   index.mdx wins via
//                                                   content-collection
//                                                   precedence over the
//                                                   plugin's injectRoute)
//   - `/guides/api/operations/{operationId}/`     → per-operation page
//   - `/guides/api/operations/tags/{slug}/`       → tag overview
// We exclude only the `operations/` subtree. The Phase 224
// pluginPagefindWeight (data-pagefind-weight="0.1" on <pre>) STAYS —
// hand-authored pages keep the soft weight reduction; auto-generated
// pages get a hard exclusion layered on top (D-23).
//
// Wired in docs/astro.config.mjs:
//   routeMiddleware: ['./src/middleware/pagefind-exclude.ts']

import { defineRouteMiddleware } from '@astrojs/starlight/route-data';

export const onRequest = defineRouteMiddleware((context) => {
  const { starlightRoute } = context.locals;
  if (context.url.pathname.startsWith('/guides/api/operations/')) {
    starlightRoute.entry.data.pagefind = false;
  }
});
```

Notes for the implementer:
- Extension MUST be `.ts` (NOT `.mjs`) — Starlight ships TypeScript types for `defineRouteMiddleware`, and the repo's `tsconfig.json` extends `astro/tsconfigs/strict` and includes `**/*` so `astro check` validates the file automatically.
- The startsWith match `'/guides/api/operations/'` is intentional — covers both per-operation pages (`/guides/api/operations/{id}/`) AND tag overviews (`/guides/api/operations/tags/{slug}/`).
- DO NOT match `'/guides/api/'` alone — that would catch the hand-authored landing page (Plan 06) and the hand-authored auth.mdx / ogc.mdx pages.
- DO NOT add additional logic for setting custom badges, weights, or other metadata. Single-purpose middleware.

Implements D-21 (auto-generated pages excluded), D-22 mechanism choice (routeMiddleware over fallbacks), D-23 (Phase 224 EC plugin still applies to hand-authored pages).
