## Exact Appended Block (as committed)

Appended verbatim after the existing line `echo "All 10 --primary-* stops in sync between marketing and docs."`:

```bash
# ── IBM Plex font-family sync (Phase 999.5 / D-03) ────────────────────────────
# Assert that all three surfaces declare IBM Plex Sans Variable as the sans font.
# The app (index.css) is the north star; marketing and docs must match.
# CI may run with only the getgeolens.com checkout — the app leg fails OPEN with
# an INFO message when geolens/frontend/src/index.css is absent.

APP_CSS="$SCRIPT_DIR/../../../geolens/frontend/src/index.css"
FONT_STRING="IBM Plex Sans Variable"

font_errors=0

# Fail-OPEN guard for app leg (RESEARCH.md "CI environment warning")
if ! test -f "$APP_CSS"; then
  echo "INFO [font]: app CSS not found at $APP_CSS — skipping app leg (CI single-repo checkout)"
  APP_CSS=""
fi

for label_css in "app:$APP_CSS" "marketing:$MARKETING_CSS" "docs:$DOCS_CSS"; do
  label="${label_css%%:*}"
  css_file="${label_css#*:}"

  # Skip the app leg when guard above cleared it
  if [ -z "$css_file" ]; then
    continue
  fi

  if ! test -f "$css_file"; then
    echo "FAIL [font]: $label CSS file not found at $css_file"
    font_errors=$((font_errors+1))
    continue
  fi

  if ! grep -qF "$FONT_STRING" "$css_file"; then
    echo "FAIL [font]: '$FONT_STRING' not found in $label ($css_file)"
    font_errors=$((font_errors+1))
  fi
done

if [ "$font_errors" -gt 0 ]; then
  echo ""
  echo "Font drift detected. All present surfaces must declare '$FONT_STRING'."
  echo "  App:       geolens/frontend/src/index.css       (skipped if not checked out)"
  echo "  Marketing: getgeolens.com/src/styles/global.css (must be present)"
  echo "  Docs:      getgeolens.com/docs/src/styles/custom.css (must be present)"
  exit 1
fi

echo "IBM Plex Sans Variable font-family declaration found in all available surfaces."
```
