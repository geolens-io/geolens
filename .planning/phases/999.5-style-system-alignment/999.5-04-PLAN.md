```css
/*
 * GeoLens Docs — Full token bridge (Phase 224 / BRAND-01..03, extended Phase 999.5).
 *
 * Sources of truth:
 *   - OKLCH primary palette: getgeolens.com/src/styles/global.css :root --primary-* (verbatim mirror)
 *   - IBM Plex fonts: matches geolens/frontend/src/index.css import paths (Phase 999.5)
 *
 * BRAND-04 drift detection: docs/scripts/check-token-sync.sh greps stops 50..900 from this
 * file and compares against marketing global.css. Stop 950 is extrapolated docs-only.
 * Phase 999.5 extends check-token-sync.sh to also assert IBM Plex font-family.
 *
 * WCAG 2.1 AA (D-03 + global.css comment block):
 *   --primary-700 oklch(0.46 0.16 250) is the minimum body-text-on-white accent.
 *   --sl-color-accent is therefore aliased to --primary-700, NOT --primary-500/600.
 *
 * Per D-05, NO Tailwind integration plugin. Raw customCss only.
 */

@import '@fontsource-variable/ibm-plex-sans/wght.css';
@import '@fontsource/ibm-plex-mono/400.css';

:root {
  /* GeoLens primary scale — mirror of getgeolens.com/src/styles/global.css :root --primary-* */
  --primary-50:  oklch(0.97 0.02 250);
  --primary-100: oklch(0.93 0.05 250);
  --primary-200: oklch(0.87 0.09 250);
  --primary-300: oklch(0.78 0.13 250);
  --primary-400: oklch(0.70 0.16 250);
  --primary-500: oklch(0.55 0.18 250);
  --primary-600: oklch(0.48 0.18 250);
  --primary-700: oklch(0.46 0.16 250);
  --primary-800: oklch(0.38 0.13 250);
  --primary-900: oklch(0.30 0.10 250);
  --primary-950: oklch(0.22 0.07 250);  /* extrapolated docs-only; skipped by drift script */

  /* IBM Plex font (Phase 999.5 — replaces Inter Variable) */
  --sl-font: 'IBM Plex Sans Variable', ui-sans-serif, system-ui, sans-serif;
  --sl-font-mono: 'IBM Plex Mono', ui-monospace, SFMono-Regular, Menlo, monospace;

  /* Starlight accent slot mappings — light mode */
  --sl-color-accent-low:  var(--primary-100);
  --sl-color-accent:      var(--primary-700);  /* WCAG AA body-link insurance */
  --sl-color-accent-high: var(--primary-800);

  /* Sidebar tint — warmer companion to cool-white body (D-02) */
  --sl-color-bg-sidebar: oklch(0.99 0.003 85);

  /* Aside accent overrides — note and tip → brand blue (D-04) */
  /* caution (orange) and danger (red) intentionally kept as Starlight defaults */
  --sl-color-blue-high:   var(--primary-800);
  --sl-color-blue:        var(--primary-700);
  --sl-color-blue-low:    var(--primary-50);
  --sl-color-purple-high: var(--primary-800);
  --sl-color-purple:      var(--primary-700);
  --sl-color-purple-low:  var(--primary-50);
}

:root[data-theme='dark'] {
  /* Starlight accent slot mappings — dark mode */
  --sl-color-accent-low:  var(--primary-900);
  --sl-color-accent:      var(--primary-400);
  --sl-color-accent-high: var(--primary-100);

  /* Aside accent overrides — dark mode (D-04) */
  --sl-color-blue-high:   var(--primary-100);
  --sl-color-blue:        var(--primary-400);
  --sl-color-blue-low:    var(--primary-900);
  --sl-color-purple-high: var(--primary-100);
  --sl-color-purple:      var(--primary-400);
  --sl-color-purple-low:  var(--primary-900);
}
```
