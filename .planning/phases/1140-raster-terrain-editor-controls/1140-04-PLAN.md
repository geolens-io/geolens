buildColormapTileUrl contract (RESEARCH Pattern 3 / Code Examples):
```
export function buildColormapTileUrl(baseUrl: string, paint: Record<string, unknown>): string {
  const colormap = paint['_colormap']; const stretch = paint['_stretch'];
  if (!colormap || colormap === 'gray') return baseUrl;          // gray = Titiler single-band default → no param
  const params = new URLSearchParams();
  params.set('colormap_name', colormap as string);
  if (typeof stretch === 'string' && stretch !== 'minmax') params.set('stretch', stretch);
  return `${baseUrl}?${params.toString()}`;
}
```
The 8 colormaps (Titiler names, UI-SPEC + Plan 01 allowlist): gray, viridis, inferno, plasma, magma, ylorrd, bugn, terrain.
