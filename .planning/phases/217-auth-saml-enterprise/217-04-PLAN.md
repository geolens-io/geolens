```typescript
import { useQuery } from '@tanstack/react-query';
import { queryKeys } from '@/lib/query-keys';
import { fetchEdition } from '@/api/edition';

export function useEdition() {
  const { data, isLoading } = useQuery({
    queryKey: queryKeys.edition.info,
    queryFn: fetchEdition,
    staleTime: Infinity,
    gcTime: Infinity,
  });

  return {
    edition: data?.edition ?? 'community',
    features: data?.features ?? EMPTY_FEATURES,
    isEnterprise: data?.edition === 'enterprise',
    isLoading,
  };
}
```

```typescript
import { apiFetch } from './client';
import type { OAuthProviderConfig } from './settings';

export interface SamlProviderConfig extends OAuthProviderConfig {
  idp_entity_id: string;
  idp_sso_url: string;
  sp_entity_id: string;
  // idp_certificate is write-only (encrypted at rest, never returned)
}

export interface SamlProviderCreateData {
  slug: string;
  display_name: string;
  provider_type: 'saml';
  default_role: string;
  group_claim?: string | null;
  group_role_mapping?: Record<string, string> | null;
  idp_entity_id: string;
  idp_sso_url: string;
  idp_certificate: string;     // PEM, write-only
  sp_entity_id: string;
  enabled?: boolean;
}

export type SamlProviderUpdateData = Partial<SamlProviderCreateData>;

export async function listSamlProviders(): Promise<SamlProviderConfig[]> {
  // Reuses /settings/oauth-providers/ and filters client-side by provider_type
  const all = await apiFetch<SamlProviderConfig[]>('/settings/oauth-providers/');
  return all.filter(p => p.provider_type === 'saml');
}

// CRUD goes through the SAME endpoints as OAuth (D-12) — re-export the wrappers
// from settings.ts; do NOT duplicate.
export {
  createOAuthProvider as createSamlProvider,
  updateOAuthProvider as updateSamlProvider,
  deleteOAuthProvider as deleteSamlProvider,
} from './settings';

export async function fetchSamlMetadata(slug: string): Promise<string> {
  // No apiFetch — metadata is XML, not JSON
  const response = await fetch(`/api/auth/saml/${slug}/metadata`);
  if (!response.ok) throw new Error(`Metadata fetch failed: ${response.status}`);
  return response.text();
}
```
