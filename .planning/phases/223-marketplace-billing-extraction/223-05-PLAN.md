```python
"""Enterprise billing — AWS Marketplace RegisterUsage hook (Phase 223 BILLING-02).

Relocated from `backend/app/core/marketplace.py` (deleted in Phase 223 Plan 03).
The 30-line `register_marketplace_usage` body now lives inside this overlay's
`MarketplaceBillingExtension._register_usage` method.

Registered with core's `BillingExtension` Protocol via the entry-point group
`geolens.extensions` (see `geolens_enterprise/__init__.py:register_extensions`).
The community core package has zero AWS Marketplace business logic after
Phase 223 — this overlay is the sole consumer of `AWS_MARKETPLACE_PRODUCT_CODE`
and the sole caller of `boto3.client("meteringmarketplace").register_usage`.

Behavior contract preserved from pre-Phase-223 core code:
  - When `AWS_MARKETPLACE_PRODUCT_CODE` is unset (the default), `on_startup`
    returns immediately. Zero AWS API calls. Zero side effects.
  - When set, the boto3 sync `register_usage` call is wrapped in
    `asyncio.to_thread` so the lifespan startup doesn't block the event loop.
  - Failures are NOT caught here — core's dispatch loop (Phase 223 D-10
    at `app/api/main.py`) wraps each extension with
    `asyncio.wait_for(timeout=10.0)` + try/except. Overlays stay simple;
    core handles isolation.
"""

from __future__ import annotations

import asyncio
import os
import uuid


class MarketplaceBillingExtension:
    """AWS Marketplace RegisterUsage hook (Phase 223 D-04 / D-13).

    Implements the `BillingExtension` Protocol from
    `app.platform.extensions.protocols`. Registered as an instance via
    `geolens_enterprise.__init__.register_extensions`.
    """

    async def on_startup(self, app) -> None:
        """Fire AWS Marketplace RegisterUsage at lifespan startup.

        Phase 223 D-13: the conditional gate (`if AWS_MARKETPLACE_PRODUCT_CODE`)
        lives in the OVERLAY, not in core dispatch. Core has no knowledge of
        the env var. Community deployments (without this overlay installed)
        get the no-op `DefaultBillingExtension.on_startup` instead.
        """
        product_code = os.environ.get("AWS_MARKETPLACE_PRODUCT_CODE")
        if not product_code:
            return

        public_key_version = int(
            os.environ.get("AWS_MARKETPLACE_PUBLIC_KEY_VERSION", "1")
        )

        await asyncio.to_thread(
            self._register_usage, product_code, public_key_version
        )

    def _register_usage(self, product_code: str, public_key_version: int) -> None:
        """Sync AWS Marketplace RegisterUsage call (relocated from core)."""
        import boto3
        import structlog

        logger = structlog.stdlib.get_logger(__name__)
        client = boto3.client("meteringmarketplace")
        resp = client.register_usage(
            ProductCode=product_code,
            PublicKeyVersion=public_key_version,
            Nonce=str(uuid.uuid4()),
        )
        logger.info(
            "AWS Marketplace metering registered",
            product_code=product_code,
            signature=resp.get("Signature", "")[:32] + "...",
        )
```
