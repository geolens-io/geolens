    ```python
    @asynccontextmanager
    async def _job_phase_session(
        job_uuid: uuid.UUID, *, phase: str
    ):
        """Two-phase session bracket for ingest workers (REMED-03 / P2-05).

        Yields (session, job) where job is None if the row vanished. Wraps:
        - async_session() lifecycle
        - SELECT IngestJob WHERE id = job_uuid
        - "vanished between phases" log + early-yield on missing job
        - rollback-on-exception
        Caller owns commits — multiple commits per phase block are normal
        (load → mark running → commit → continue → commit again).
        Enforces the #100 greenlet rule by keeping the session lifetime
        scoped to the `async with` — long-running CPU/subprocess work
        MUST happen OUTSIDE this block, never inside.
        """
        from app.core.db import async_session
        from app.platform.jobs.models import IngestJob
        from sqlalchemy import select

        async with async_session() as session:
            result = await session.execute(
                select(IngestJob).where(IngestJob.id == job_uuid)
            )
            job = result.scalar_one_or_none()
            if job is None:
                structlog.get_logger().warning(
                    "Ingest job not found in phase, skipping",
                    job_id=str(job_uuid),
                    phase=phase,
                )
                yield session, None
                return
            try:
                yield session, job
            except Exception:
                await session.rollback()
                raise
    ```
