```bash
#!/usr/bin/env bash
# Phase 1061 SEC-S06: generate per-deploy random credentials for the demo
# overlay. Writes .env.demo (gitignored). Refuses to overwrite without --force.
#
# Usage:
#   scripts/init-demo-env.sh           # write fresh .env.demo (refuses if file exists)
#   scripts/init-demo-env.sh --force   # overwrite existing .env.demo
#
# Then start the stack:
#   docker compose -f docker-compose.yml -f docker-compose.demo.yml --env-file .env.demo up -d
set -euo pipefail

FORCE=false
if [[ "${1:-}" == "--force" ]]; then
  FORCE=true
fi

ROOT="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"
TARGET="$ROOT/.env.demo"

if [[ -f "$TARGET" && "$FORCE" != "true" ]]; then
  echo "Error: $TARGET already exists. Pass --force to overwrite." >&2
  exit 1
fi

command -v openssl >/dev/null 2>&1 || {
  echo "Error: openssl not found on PATH (required for credential generation)." >&2
  exit 1
}

# 48 bytes base64 → 64 chars; strip newlines, '/', '=' to avoid env-file parsing issues
gen() {
  local bytes=$1
  openssl rand -base64 "$bytes" | tr -d '\n=+/' | head -c $((bytes * 4 / 3))
}

JWT=$(gen 48)
PG=$(gen 24)
ADM=$(gen 18)
MINIO_USER=$(gen 12)
MINIO_PASS=$(gen 24)

TMP=$(mktemp)
trap 'rm -f "$TMP"' EXIT

cat > "$TMP" <<EOF
# Generated by scripts/init-demo-env.sh on $(date -u +%Y-%m-%dT%H:%M:%SZ)
# Per-deploy random credentials — DO NOT COMMIT (covered by .gitignore).

# --- Database ---
POSTGRES_DB=geolens
POSTGRES_USER=geolens
POSTGRES_PASSWORD=$PG

# --- Authentication ---
JWT_SECRET_KEY=$JWT
GEOLENS_ADMIN_USERNAME=admin
GEOLENS_ADMIN_PASSWORD=$ADM

# --- Demo opt-in (required by validate_demo_credentials_guard) ---
GEOLENS_DEMO_MODE=true

# --- MinIO (cloud-dev profile only) ---
MINIO_ROOT_USER=$MINIO_USER
MINIO_ROOT_PASSWORD=$MINIO_PASS

# --- Logging ---
LOG_JSON=true
LOG_LEVEL=INFO

# --- Upload (reduced for demo safety) ---
UPLOAD_MAX_SIZE_MB=50

# --- Host Ports ---
DB_PORT=5432
API_PORT=8000
FRONTEND_PORT=8080

# --- Demo Reset ---
RESET_INTERVAL_HOURS=24
EOF

mv "$TMP" "$TARGET"
chmod 600 "$TARGET"

echo "Generated $TARGET"
echo "Admin password (capture this — it is the only output): $ADM"
```

```python
@model_validator(mode="after")
def validate_demo_credentials_guard(self) -> "Settings":
    """Refuse to boot with known-public .env.demo defaults.

    Phase 268 H-19 (original): refused boot with .env.demo defaults UNLESS
    GEOLENS_DEMO_MODE=true was explicitly set.

    Phase 1061 SEC-S06 (extended): refuses boot with the LITERAL committed
    .env.demo values REGARDLESS of GEOLENS_DEMO_MODE. The audit's headline
    scenario is an operator deploying docker-compose.demo.yml verbatim to a
    public-internet host with GEOLENS_DEMO_MODE=true — that previously
    succeeded, leaking known-public JWT / admin credentials. Now the operator
    must first run scripts/init-demo-env.sh to generate per-deploy random
    values; the literal committed strings are always refused.

    Note: GEOLENS_DEMO_MODE=true still relaxes other policies (e.g., admin
    seeding) — this validator only blocks the specific literal strings.
    """
    jwt_value = self.jwt_secret_key.get_secret_value()
    admin_value = self.geolens_admin_password.get_secret_value()
    pg_value = self.postgres_password.get_secret_value()

    hint = (
        " Run `scripts/init-demo-env.sh` to generate per-deploy random "
        "credentials, or set the value manually with `openssl rand -hex 32`."
    )

    if jwt_value == DEMO_JWT_SECRET:
        raise ValueError(
            "JWT_SECRET_KEY is set to the known-public .env.demo template "
            "value. This value is committed to the public repository and "
            "anyone with read access can forge JWTs against this deployment."
            + hint
        )

    if admin_value == DEMO_ADMIN_PASSWORD:
        raise ValueError(
            "GEOLENS_ADMIN_PASSWORD is set to the known-public .env.demo "
            "template value (`demodemo`). This is committed to the public "
            "repository and anyone with read access can log in as admin."
            + hint
        )

    if pg_value == DEMO_POSTGRES_PASSWORD:
        raise ValueError(
            "POSTGRES_PASSWORD is set to the known-public .env.demo template "
            "value. This is committed to the public repository."
            + hint
        )

    return self
```

This eliminates the demo-mode bypass entirely. Boot succeeds only when ALL three credentials are NOT the literal committed strings. `GEOLENS_DEMO_MODE=true` still controls other behaviors (admin auto-seeding, reset interval) but no longer relaxes this guard.
