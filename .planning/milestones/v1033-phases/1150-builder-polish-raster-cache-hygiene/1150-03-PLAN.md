```python
def test_band_stats_cache_eviction():
    from cachetools import LRUCache
    # Directly test LRUCache eviction semantics for the cache we use
    cache: LRUCache[str, list[dict] | None] = LRUCache(maxsize=256)
    for i in range(257):
        cache[f"path-{i}"] = [{"b1": i}]
    assert len(cache) == 256
    # path-0 (oldest) should have been evicted
    assert "path-0" not in cache
    # path-256 (newest) should still be present
    assert "path-256" in cache
```
