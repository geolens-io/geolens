```typescript
// Existing isUrl helper (FeaturePopup.tsx:38-41) — KEEP for backward compat:
function isUrl(value: unknown): value is string {
  if (typeof value !== 'string') return false;
  return value.startsWith('http://') || value.startsWith('https://');
}

// Existing ValueDisplay (FeaturePopup.tsx:246-289) — handles entire-value-is-URL.
// New behavior: when value is a string that contains URLs (not necessarily ONLY a URL),
// split it into text/url segments and render each segment.
// When the URL classifies as media, render media inline.

// New helper module shape (popup-rich-text.ts):
export type UrlKind = 'image' | 'video' | 'youtube' | 'other';

export interface UrlClassification {
  kind: UrlKind;
  /** For YouTube: the canonical embed URL. For others: the original URL. */
  srcUrl: string;
}

export type RichSegment = { kind: 'text'; value: string } | { kind: 'url'; value: string };

/** Match http:// or https:// URLs in arbitrary text. Excludes javascript: / data: URLs (XSS gate). */
export function detectUrls(text: string): RegExpMatchArray | null;

/** Split text into ordered text + URL segments. detectUrls match boundaries define the split. */
export function splitTextWithUrls(text: string): RichSegment[];

/** Classify a URL as image, video, YouTube embed, or other (plain link). Case-insensitive extension match. */
export function classifyUrl(url: string): UrlClassification;

/** Extract the canonical YouTube embed URL from a watch / shortlink / shorts URL. Returns null when not a recognised YouTube URL. */
export function normalizeYouTubeEmbed(url: string): string | null;
```

Detection rules:
- URL regex: `/https?:\/\/[^\s<>"']+/gi` (intentionally exclude angle brackets, quotes, whitespace).
- javascript: / data: / vbscript: URLs MUST be excluded from detectUrls — XSS gate (React JSX text-node escaping is the secondary gate; defense-in-depth).
- Image extensions: .jpg / .jpeg / .png / .gif / .webp / .svg (case-insensitive). Allow optional query string (`?v=2`) and trailing fragments (`#crop`).
- Video extensions: .mp4 / .webm / .mov.
- YouTube patterns: `youtube.com/watch?v=<ID>`, `youtu.be/<ID>`, `youtube.com/shorts/<ID>`, `youtube.com/embed/<ID>` — ID is `[A-Za-z0-9_-]{11}`. Return `https://www.youtube.com/embed/<ID>` as the canonical srcUrl.
