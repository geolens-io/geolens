3. `useChatActionStaging(dispatch)` hook:
```ts
export function useChatActionStaging(
  dispatch: (action: PendingAction) => void,
): ChatActionStaging {
  const [pendingActions, setPendingActions] = useState<PendingAction[]>([]);
  const dispatchRef = useRef(dispatch);
  dispatchRef.current = dispatch;  // mirror latest dispatch to avoid stale closure

  const push = useCallback((action: PendingAction) => {
    setPendingActions((prev) => [...prev, action]);
  }, []);

  const rejectAll = useCallback(() => {
    setPendingActions([]);
  }, []);

  const acceptAll = useCallback(async () => {
    // Snapshot the current buffer THEN clear; otherwise interleaving setState
    // updates can race with the dispatch loop.
    let snapshot: PendingAction[] = [];
    setPendingActions((prev) => { snapshot = prev; return []; });
    for (const action of snapshot) {
      dispatchRef.current(action);
    }
  }, []);

  const rejectOne = useCallback((index: number) => {
    setPendingActions((prev) => {
      if (index < 0 || index >= prev.length) return prev;
      const next = prev.slice();
      next.splice(index, 1);
      return next;
    });
  }, []);

  const acceptOne = useCallback(async (index: number) => {
    let action: PendingAction | undefined;
    setPendingActions((prev) => {
      if (index < 0 || index >= prev.length) return prev;
      action = prev[index];
      const next = prev.slice();
      next.splice(index, 1);
      return next;
    });
    // queueMicrotask guarantees dispatch fires AFTER React commits the splice
    // — keeps the rejectAll-style "byte-equal state on reject" contract clean.
    await Promise.resolve();
    if (action) dispatchRef.current(action);
  }, []);

  return { pendingActions, push, acceptAll, rejectAll, acceptOne, rejectOne };
}
```

Imports needed: `useCallback, useRef, useState` from 'react'; `ChatAction` type from '@/types/api'.
