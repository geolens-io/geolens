    ```ts
    import { describe, it, expect } from 'vitest';

    /**
     * Phase 1136 EDITOR Pitfall #9 grep guard (phase-scoped).
     *
     * Asserts that the 5 editor source files modified during Phase 1136
     * contain ZERO direct `map.setPaintProperty` / `map.setLayoutProperty`
     * callsites. All paint/layout mutations from these files MUST route
     * through the v1026 owned-property contract:
     *   - paint: `onPaintProp(key, value)` → adapter `syncPaint` → `*_OWNED_PAINT_PROPERTIES`
     *   - layout: `onLayoutChange(layerId, layout)` → reconciler hook `handleLayoutChange`
     *             (documented exception per Phase 1133 WALK-04 audit)
     *
     * This guard is SCOPED to the 5 phase-1136 files. The broader baseline
     * of documented exceptions (label-layer-utils, ViewerMap.tsx, reconciler
     * hooks, basemap-style-mutation.ts) is the responsibility of Phase 1133
     * WALK-04's audit table, not this guard.
     */

    const FRONTEND_ROOT = resolve(__dirname, '..', '..', '..', '..');

    const WATCHED_FILES = [
      'src/components/builder/LayerStyleEditor/RasterEditor.tsx',
      'src/components/builder/LayerStyleEditor/LineEditor.tsx',
      'src/components/builder/LayerStyleEditor/FillEditor.tsx',
      'src/components/builder/BasemapGroupEditorScene.tsx',
      'src/components/builder/BasemapSublayerEditorScene.tsx',
    ];

    function stripComments(src: string): string {
      // Drop // line comments (line-anchored)
      let s = src.replace(/(^|[^:])\/\/[^\n]*/g, '$1');
      // Drop /* ... */ block comments (non-greedy, multi-line)
      s = s.replace(/\/\*[\s\S]*?\*\//g, '');
      return s;
    }

    function countMatches(src: string, pattern: RegExp): number {
      const stripped = stripComments(src);
      const matches = stripped.match(pattern);
      return matches ? matches.length : 0;
    }

    describe('Phase 1136 EDITOR Pitfall #9 grep guard', () => {
      it.each(WATCHED_FILES)(
        '%s contains ZERO direct map.setPaintProperty callsites',
        (relPath) => {
          const fullPath = resolve(FRONTEND_ROOT, relPath);
          expect(existsSync(fullPath), `Watched file missing: ${relPath}`).toBe(true);
          const src = readFileSync(fullPath, 'utf-8');
          const count = countMatches(src, /map\.setPaintProperty/g);
          expect(
            count,
            `Phase 1136 Pitfall #9 violation: ${relPath} contains ${count} direct map.setPaintProperty callsite(s). Route paint writes through onPaintProp + adapter OWNED_PAINT_PROPERTIES instead.`,
          ).toBe(0);
        },
      );

      it.each(WATCHED_FILES)(
        '%s contains ZERO direct map.setLayoutProperty callsites',
        (relPath) => {
          const fullPath = resolve(FRONTEND_ROOT, relPath);
          expect(existsSync(fullPath), `Watched file missing: ${relPath}`).toBe(true);
          const src = readFileSync(fullPath, 'utf-8');
          const count = countMatches(src, /map\.setLayoutProperty/g);
          expect(
            count,
            `Phase 1136 Pitfall #9 violation: ${relPath} contains ${count} direct map.setLayoutProperty callsite(s). Route layout writes through onLayoutChange + adapter OWNED_LAYOUT_PROPERTIES instead.`,
          ).toBe(0);
        },
      );
    });
    ```
